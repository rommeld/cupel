use gpui::{AppContext, Application, KeyBinding, WindowOptions};

use cupel::actions::{
    Commit, CollapseAllDiffHunks, ExpandAllDiffHunks, FocusDiffPanel, FocusGitPanel, GoToHunk,
    GoToPrevHunk, Quit, RevertHunk, SelectNext, SelectPrev, StageAll, StageHunk, ToggleHunkDiff,
    ToggleStaging, UnstageAll, UnstageHunk,
};
use cupel::assets::Assets;
use cupel::git::types::GitPanelSettings;
use cupel::theme::Theme;
use cupel::ui::app::AppView;

fn main() {
    Application::new()
        .with_assets(Assets)
        .run(|cx| {
            cx.set_global(Theme::default());
            cx.set_global(GitPanelSettings::default());

            cx.bind_keys([
                KeyBinding::new("cmd-q", Quit, None),
                // Git panel
                KeyBinding::new("enter", ToggleStaging, Some("git_panel")),
                KeyBinding::new("down", SelectNext, Some("git_panel")),
                KeyBinding::new("up", SelectPrev, Some("git_panel")),
                KeyBinding::new("cmd-shift-a", StageAll, Some("git_panel")),
                KeyBinding::new("cmd-shift-u", UnstageAll, Some("git_panel")),
                KeyBinding::new("cmd-enter", Commit, Some("git_panel")),
                // Diff panel
                KeyBinding::new("down", SelectNext, Some("diff_panel")),
                KeyBinding::new("up", SelectPrev, Some("diff_panel")),
                KeyBinding::new("enter", ToggleHunkDiff, Some("diff_panel")),
                KeyBinding::new("j", GoToHunk, Some("diff_panel")),
                KeyBinding::new("k", GoToPrevHunk, Some("diff_panel")),
                KeyBinding::new("s", StageHunk, Some("diff_panel")),
                KeyBinding::new("u", UnstageHunk, Some("diff_panel")),
                KeyBinding::new("x", RevertHunk, Some("diff_panel")),
                KeyBinding::new("cmd-shift-e", ExpandAllDiffHunks, Some("diff_panel")),
                KeyBinding::new("cmd-shift-c", CollapseAllDiffHunks, Some("diff_panel")),
                // Panel switching
                KeyBinding::new("tab", FocusDiffPanel, Some("git_panel")),
                KeyBinding::new("tab", FocusGitPanel, Some("diff_panel")),
            ]);

            cx.on_action(|_: &Quit, cx| {
                cx.quit();
            });

            cx.open_window(WindowOptions::default(), |window, cx| {
                cx.new(|cx| AppView::new(window, cx))
            })
            .expect("failed to open window");
        });
}
